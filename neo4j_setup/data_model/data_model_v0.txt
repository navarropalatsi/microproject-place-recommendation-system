MERGE (u:User)
MERGE (p:Place)
MERGE (f:Feature)
MERGE (c:Category)

MERGE (u)-[:NEEDS_FEATURE]->(f)

MERGE (p)-[:HAS_FEATURE]->(f)

MERGE (u)-[v:VISITED]->(p)
ON CREATE
    SET v.createdAt = timestamp()

MERGE (u)-[r:RATED]->(p)
ON CREATE
    SET r.createdAt = timestamp()

MERGE (p)-[:IN_CATEGORY]->(c)

CREATE CONSTRAINT RATED_rating
FOR ()-[r:RATED]->() REQUIRE r.rating IS NOT NULL

CREATE CONSTRAINT user_name
FOR (u:User) REQUIRE u.name IS NOT NULL

CREATE CONSTRAINT user_userId
FOR (u:User) REQUIRE u.userId IS NOT NULL

CREATE CONSTRAINT user_userId

CREATE CONSTRAINT place_placeId
